Create

   SELECT * FROM table WHERE column = 'key';
Read Some
   SELECT * FROM table WHERE column = 'search word';
Read All
   SELECT * FROM table;

   UPDATE table SET column1 = 'value',column2 = 'value'WHERE column = 'key';
